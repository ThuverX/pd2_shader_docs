Vertex Shader

float GLOBAL_fov_ratio : register(vs_3_0, c8);
float camera_near_range : register(vs_3_0, c7);
column_major float4x4 view_proj_matrix : register(vs_3_0, c0);
column_major float4x4 world_matrix : register(vs_3_0, c4);
struct VertexMain_Input
{
    float4 position : POSITION;
    float4 normal : NORMAL;
    float4 texcoord : TEXCOORD;
    float4 texcoord1 : TEXCOORD1;
    float4 tangent : TANGENT;
    float4 binormal : BINORMAL;
};

struct VertexMain_Output
{
    float4 position : POSITION;
    float4 texcoord : TEXCOORD;
    float4 texcoord1 : TEXCOORD1;
    float3 texcoord2 : TEXCOORD2;
    float3 texcoord3 : TEXCOORD3;
    float3 texcoord4 : TEXCOORD4;
};

VertexMain_Output VertexMain(VertexMain_Input i)
{
    VertexMain_Output o;
    float3 temp0, temp1;
    // def c9, 2, -1, 0, 0
    // dcl_position v0
    // dcl_normal v1
    // dcl_texcoord v2
    // dcl_texcoord1 v3
    // dcl_tangent v4
    // dcl_binormal v5
    // dcl_position o0
    // dcl_texcoord o2
    // dcl_texcoord1 o3
    // dcl_texcoord2 o4.xyz
    // dcl_texcoord3 o5.xyz
    // dcl_texcoord4 o6.xyz
    // dp4 o0.x, v0, c0
    o.position.x = dot(i.position, (view_proj_matrix._m00_m10_m20_m30));
    // dp4 o0.y, v0, c1
    o.position.y = dot(i.position, (view_proj_matrix._m01_m11_m21_m31));
    // dp4 o0.w, v0, c3
    o.position.w = dot(i.position, (view_proj_matrix._m03_m13_m23_m33));
    // rsq r0.x, c8.x
    temp0.x = 1 / sqrt(GLOBAL_fov_ratio.x);
    // dp4 r0.y, v0, c2
    temp0.y = dot(i.position, (view_proj_matrix._m02_m12_m22_m32));
    // mad o3.w, r0.y, r0.x, c7.x
    o.texcoord1.w = temp0.y * temp0.x + camera_near_range.x;
    // mov o0.z, r0.y
    o.position.z = temp0.y;
    // mad r0.xyz, v4, c9.x, c9.y
    temp0.xyz = i.tangent.xyz * float3(2, 2, 2) + float3(-1, -1, -1);
    // dp3 o5.x, r0, c5
    o.texcoord3.x = dot(temp0.xyz, (world_matrix._m01_m11_m21_m31).xyz);
    // dp3 o6.x, r0, c6
    o.texcoord4.x = dot(temp0.xyz, (world_matrix._m02_m12_m22_m32).xyz);
    // mad r0.xyz, v5, c9.x, c9.y
    temp0.xyz = i.binormal.xyz * float3(2, 2, 2) + float3(-1, -1, -1);
    // dp3 o5.y, -r0, c5
    o.texcoord3.y = dot(-temp0.xyz, (world_matrix._m01_m11_m21_m31).xyz);
    // dp3 o6.y, -r0, c6
    o.texcoord4.y = dot(-temp0.xyz, (world_matrix._m02_m12_m22_m32).xyz);
    // mad r0.xyz, v1, c9.x, c9.y
    temp0.xyz = i.normal.xyz * float3(2, 2, 2) + float3(-1, -1, -1);
    // dp3 r1.x, r0, c4
    temp1.x = dot(temp0.xyz, (world_matrix._m00_m10_m20_m30).xyz);
    // dp3 r1.y, r0, c5
    temp1.y = dot(temp0.xyz, (world_matrix._m01_m11_m21_m31).xyz);
    // dp3 r1.z, r0, c6
    temp1.z = dot(temp0.xyz, (world_matrix._m02_m12_m22_m32).xyz);
    // dp3 r0.x, r1, r1
    temp0.x = dot(temp1.xyz, temp1.xyz);
    // rsq r0.x, r0.x
    temp0.x = 1 / sqrt(temp0.x);
    // mul o4.xyz, r0.x, r1
    o.texcoord2 = temp0.x * temp1;
    // mov o5.z, r1.y
    o.texcoord3.z = temp1.y;
    // mov o6.z, r1.z
    o.texcoord4.z = temp1.z;
    // dp4 o3.x, v0, c4
    o.texcoord1.x = dot(i.position, (world_matrix._m00_m10_m20_m30));
    // dp4 o3.y, v0, c5
